0 : Math.max(0, Number(d.points)||0); }); state.history.push({ round:roundNumber(), date:new Date().toISOString(), scores }); saveState(); const afterLeader=sortedPlayers()[0]; const champ=championCandidate(); if(champ){ lastFeedback={type:'champion', player:champ, score:scoreOf(champ.id)}; } else if(beforeLeader && afterLeader && beforeLeader.id!==afterLeader.id){ lastFeedback={type:'newLeader', player:afterLeader, score:scoreOf(afterLeader.id)}; } else { lastFeedback={type:'sameLeader', player:afterLeader, score:scoreOf(afterLeader.id)}; } renderFeedback(); go('feedback'); } function renderFeedback(){ const title=document.getElementById('feedbackTitle'), text=document.getElementById('feedbackText'), card=document.getElementById('feedbackCard'), finish=document.getElementById('finishGameBtn'); if(!lastFeedback) return; const p=lastFeedback.player; if(lastFeedback.type==='champion'){ title.textContent='Temos um campeão!'; text.textContent='Maior pontuação acima de 200 é a campeã.'; finish.classList.remove('hidden'); } else if(lastFeedback.type==='newLeader'){ title.textContent='Nova liderança!'; text.textContent=`${p.name} assumiu o primeiro lugar.`; finish.classList.add('hidden'); } else { title.textContent='Líder mantido!'; text.textContent=`${p.name} continua na liderança.`; finish.classList.add('hidden'); } card.innerHTML=`
${escapeHtml(p.name)}
${lastFeedback.score}
pontos
`; } function finishGame(){ const champ=championCandidate(); if(!champ) return; state.champions.unshift({ id:uid(), playerName:champ.name, score:scoreOf(champ.id), date:new Date().toISOString(), round:state.history.length }); state.gameEnded=true; saveState(); renderChampions(); go('champions'); } function renderHistory(){ const table=document.getElementById('historyTable'); if(!state.history.length){ table.innerHTML='Nenhuma rodada salva.'; return; } const players=state.players; const header=`Rodada${players.map(p=>`${escapeHtml(p.name)}`).join('')}`; const body=`${[...state.history].reverse().map(r=>`${r.round}${players.map(p=>`${r.scores[p.id]===0?'💣':(r.scores[p.id]??0)}`).join('')}`).join('')}`; const foot=`Total${players.map(p=>`${scoreOf(p.id)}`).join('')}`; table.innerHTML=header+body+foot; } function renderChampions(){ const box=document.getElementById('championsList'); box.innerHTML=''; if(!state.champions.length){ box.innerHTML='
